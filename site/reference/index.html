Provide functions to run your site connectivity application

The module contains the following functions to run the main: - _get_websites_urls(user_args:any) -> List[str]: Return a list of websites urls _ read_urls_from_file(file:str) -> List[str]: Return a list of urls from a file _ asynchronous_check(urls:List[str],time_out:float) -> None: Display the results of asynchronous option _ synchronous_check(urls:List,timeout:float) -> None: Display the results of asynchronous option

def main():
    """ Run RP Checker"""
    user_args = read_user_cli_args()
    urls = _get_websites_urls(user_args)
    if not urls:
        logger.error("No URLs to check!")
        sys.exit(1)
    time_out = float(input("Enter time out: "))
    if user_args.asynchronous:
        asyncio.run(_asynchronous_check(urls,time_out))
    else:
        _synchronous_check(urls,timeout)
